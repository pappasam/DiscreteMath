	\item[17] \hfill \\
		\begin{description}
			\item[a] \hfill \\
				\begin{tabular}{c | c || c}
					$p$ & % 1
					$\lnot p$ & % 2
					$\lnot \lnot p$ \\ % 3
					\hline
					%   2   3
					F & T & F \\
					T & F & T \\
				\end{tabular}
				\begin{proof}
					Since $p$ and $\lnot \lnot p$ have the same truth table,
					$p \equiv \lnot \lnot p$
				\end{proof}
			\item[b] \hfill \\
				\begin{tabular}{c | c | c | c}
					$\underline{t}$ & % 1
					$\underline{f}$ & % 2
					$\lnot \underline{t}$ & % 3
					$\lnot \underline{f}$ \\ % 4
					\hline
					T & F & F & T \\
				\end{tabular}
				\begin{proof}
					Since the truth tables for $\lnot \underline{t}$ and $\underline{f}$
					are equivalent $\lnot \underline{t} \equiv \underline{f}$. Similarly, since
					the truth tables for $\lnot \underline{f}$ and $\underline{t}$ are equivalent,
					$\lnot \underline{f} \equiv \underline{t}$.
				\end{proof}
		\end{description}
